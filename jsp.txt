-- üß™ FULL PROFILE DATA DUMPER
-- ===============================================

local ReplicatedStorage = game:GetService("ReplicatedStorage")
local Networker = ReplicatedStorage.Shared.Packages.Networker

task.wait(1)

local GetProfileDataRF = Networker["RF/GetProfileData"]

local success, profileData = pcall(function()
    return GetProfileDataRF:InvokeServer()
end)

if not success then
    warn("‚ùå RF/GetProfileData a √©chou√©")
    return
end

if typeof(profileData) ~= "table" then
    warn("‚ùå ProfileData n'est pas une table")
    return
end

print("========== üîç PROFILE DATA ROOT KEYS ==========")

for k, v in pairs(profileData) do
    print("üóÇÔ∏è", k, "| type:", typeof(v))
end

print("========== üîç PROFILE DATA DEEP DUMP ==========")

local function dumpTable(tbl, indent, visited)
    visited = visited or {}
    if visited[tbl] then
        print(indent .. "‚Ü©Ô∏è (d√©j√† visit√©)")
        return
    end
    visited[tbl] = true

    for k, v in pairs(tbl) do
        local prefix = indent .. tostring(k) .. " (" .. typeof(v) .. ")"

        if typeof(v) == "table" then
            print(prefix)
            dumpTable(v, indent .. "   ", visited)
        else
            print(prefix .. " = " .. tostring(v))
        end
    end
end

dumpTable(profileData, "")

print("========== ‚úÖ FIN DUMP PROFILE DATA ==========")
